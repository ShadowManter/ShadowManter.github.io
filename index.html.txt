        // Database simulation object structure mapping distinct accounts
        const userDatabase = {
            user1: {
                name: "Alex Rivera",
                handle: "@alex_tracks",
                avatar: "🎧",
                logs: [
                    { type: "movie", title: "Inception", marker: "Nolan Director Cut", rating: "★★★★★", note: "Mind-bending visual setups. The structural layers hold up incredibly well on rewatch." },
                    { type: "music", title: "Blinding Lights", marker: "The Weeknd", rating: "★★★★☆", note: "Exceptional retro synthesiser pacing and vocal arrangement patterns." },
                    { type: "game", title: "Elden Ring", marker: "FromSoftware", rating: "★★★★★", note: "Massive scale open world landscape navigation design. Brutal mechanics but deeply rewarding loop." }
                ]
            },
            user2: {
                name: "Sarah Jenkins",
                handle: "@sarah_reviews",
                avatar: "🎬",
                logs: [
                    { type: "movie", title: "Interstellar", marker: "Paramount+", rating: "★★★★★", note: "Stunning scientific conceptual frameworks anchored by brilliant organ scores." },
                    { type: "music", title: "Bohemian Rhapsody", marker: "Queen", rating: "★★★★★", note: "Incredible genre-shifting structure. A masterclass in operatic vocal design." },
                    { type: "game", title: "Hades", marker: "Supergiant Games", rating: "★★★★☆", note: "Flawless isometric rogue-like action flows combined with rich interactive narrative beats." }
                ]
            }
        };

        // Interface render routine
        function renderFeed(userId) {
            const user = userDatabase[userId];
            document.getElementById("profileName").innerText = user.name;
            document.getElementById("profileHandle").innerText = user.handle;
            document.getElementById("profileAvatar").innerText = user.avatar;

            const feedContainer = document.getElementById("activityFeed");
            feedContainer.innerHTML = ""; // Clear existing records

            user.logs.forEach(log => {
                let iconClass = "";
                let typeEmoji = "";
                if (log.type === "movie") { iconClass = "icon-movie"; typeEmoji = "🎬"; }
                if (log.type === "music") { iconClass = "icon-music"; typeEmoji = "🎵"; }
                if (log.type === "game") { iconClass = "icon-game"; typeEmoji = "🎮"; }

                const cardHtml = `
                    <article class="log-card">
                        <div class="media-icon-indicator ${iconClass}">${typeEmoji}</div>
                        <div class="log-meta">
                            <h3>${log.title} <span style="color: #ffd32a; font-size: 1rem; margin-left: 10px;">${log.rating}</span></h3>
                            <div class="log-subtext">${log.type.toUpperCase()} • ${log.marker}</div>
                            <p class="review-text">${log.note}</p>
                        </div>
                    </article>
                `;
                feedContainer.innerHTML += cardHtml;
            });
        }

        // Dropdown selection routing trigger
        function switchUserProfile() {
            const selectedUser = document.getElementById("userSelect").value;
            renderFeed(selectedUser);
        }

        // Run baseline setup render immediately on load
        renderFeed("user1");
    </script>
